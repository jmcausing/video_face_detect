
# Steps to setup video_face_detect.py in  Orange Pi One Armbian 5.15.63-sunxi
# Update repo and upgrade packages
sudo apt-get update -y && sudo apt-get upgrade -y

# Increase Swap
sudo swapoff -a
sudo dd if=/dev/zero of=/swapfile bs=4K count=704850 
sudo mkswap /swapfile
sudo swapon /swapfile

# Install required packages
sudo apt-get install -y build-essential cmake gfortran git wget curl graphicsmagick libgraphicsmagick1-dev libavcodec-dev libavformat-dev libboost-all-dev libgtk2.0-dev libjpeg-dev liblapack-dev libswscale-dev pkg-config python3-dev python3-numpy python3-pip zip
sudo apt-get clean -y

# This might make the image processing faster
sudo apt-get install -y libopenblas-dev

# Install dlib via pip
screen -S pip
mkdir $HOME/tmp
TMPDIR=$HOME/tmp pip3 install dlib -vv

# Download video face detect repo
cd ~
git clone https://github.com/jmcausing/video_face_detect.git video_face_detect
cd video_face_detect/

# Alternative for opencv precompiled - not yet tested
# https://github.com/dlime/Faster_OpenCV_4_Raspberry_Pi

# Upgrade pip and install pip requirementts
pip3 install --upgrade pip
TMPDIR=~/tmp pip3 install -r ~/video_face_detect/requirements.txt -vv
